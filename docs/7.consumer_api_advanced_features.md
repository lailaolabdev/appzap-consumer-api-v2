## Feature 1: AI Genie Search

### Product Requirements

**User Story:**
> As a Consumer App user, I want to search for restaurants using natural language (e.g., "Find me a rooftop bar with cheap beer near Patuxay") so that I can quickly find what I'm looking for without using complex filters.

**Key Features:**
1. Natural language query parsing
2. Intent recognition (venue type, price, cuisine, location, amenities)
3. Semantic search across restaurant data
4. Ranked results with match confidence scores
5. Query refinement suggestions
6. Search analytics for continuous improvement

---

### API Specification

#### POST /v1/eats/search/smart

**Purpose**: AI-powered natural language restaurant search using LLM

**Implementation Type**: 🟠 **Orchestration**

**Authentication**: Required (Bearer token)

**Rate Limiting**: 
- Authenticated users: 30 requests/minute
- Due to AI API costs, implement stricter limits

---

#### Request Specification

**Headers**:
```
Authorization: Bearer {access_token}
Content-Type: application/json
```

**Request Body**:
```json
{
  "query": "Find me a rooftop bar with cheap beer near Patuxay",
  "userLocation": {
    "latitude": 17.9757,
    "longitude": 102.6331
  },
  "mode": "eats",
  "maxResults": 10,
  "filters": {
    "openNow": true,
    "acceptsReservations": false
  }
}
```

**Field Descriptions**:

| Field | Type | Required | Description |
|-------|------|----------|-------------|
| `query` | string | ✅ Yes | Natural language search query (max 200 chars) |
| `userLocation` | object | ❌ No | User's GPS coordinates for proximity ranking |
| `userLocation.latitude` | number | ❌ No | Latitude (-90 to 90) |
| `userLocation.longitude` | number | ❌ No | Longitude (-180 to 180) |
| `mode` | string | ❌ No | Search mode: `"eats"` or `"live"` (default: `"eats"`) |
| `maxResults` | number | ❌ No | Maximum results to return (1-20, default: 10) |
| `filters` | object | ❌ No | Additional hard filters |
| `filters.openNow` | boolean | ❌ No | Only show currently open restaurants |
| `filters.acceptsReservations` | boolean | ❌ No | Only show restaurants accepting bookings |

```javascript
// src/controllers/genie.controller.js

const genieService = require('../services/genie.service');
const logger = require('../utils/logger');
const { ValidationError } = require('../utils/errors');

/**
 * AI-powered natural language restaurant search
 */
exports.smartSearch = async (req, res) => {
  try {
    const { query, userLocation, mode = 'eats', maxResults = 10, filters = {} } = req.body;
    const userId = req.user.id;
    
    // Validate query
    if (!query || query.trim().length < 3) {
      throw new ValidationError('Query must be at least 3 characters');
    }
    
    if (query.length > 200) {
      throw new ValidationError('Query must be less than 200 characters');
    }
    
    // Validate maxResults
    if (maxResults < 1 || maxResults > 20) {
      throw new ValidationError('maxResults must be between 1 and 20');
    }
    
    // Check rate limit (additional check for AI API costs)
    const rateLimitKey = `genie:ratelimit:${userId}`;
    const currentCount = await redis.incr(rateLimitKey);
    
    if (currentCount === 1) {
      await redis.expire(rateLimitKey, 60); // 1 minute window
    }
    
    if (currentCount > 30) {
      return res.status(429).json({
        error: 'RATE_LIMIT_EXCEEDED',
        message: 'AI search rate limit exceeded. Please try again in 60 seconds.',
        retryAfter: 60,
        remainingQuota: 0
      });
    }
    
    // Log search query for analytics
    logger.info('AI search query', {
      userId,
      query,
      mode,
      hasLocation: !!userLocation
    });
    
    // Execute AI search
    const startTime = Date.now();
    const searchResults = await genieService.search({
      query,
      userLocation,
      mode,
      maxResults,
      filters,
      userId
    });
    
    const processingTime = Date.now() - startTime;
    
    // Save search to database for analytics
    await SearchLog.create({
      userId,
      query,
      mode,
      resultsCount: searchResults.results.length,
      processingTime,
      queryIntent: searchResults.queryIntent,
      aiModel: searchResults.searchMetadata.aiModel,
      confidenceScore: searchResults.searchMetadata.confidenceScore,
      userLocation
    });
    
    // Return results
    return res.json(searchResults);
    
  } catch (error) {
    logger.error('AI search error', { error: error.message, stack: error.stack });
    
    if (error.name === 'ValidationError') {
      return res.status(400).json({
        error: 'VALIDATION_ERROR',
        message: error.message
      });
    }
    
    if (error.message.includes('OpenAI') || error.message.includes('API')) {
      return res.status(503).json({
        error: 'AI_SERVICE_UNAVAILABLE',
        message: 'AI search is temporarily unavailable. Please use regular search.',
        fallbackEndpoint: '/v1/eats/restaurants'
      });
    }
    
    return res.status(500).json({
      error: 'INTERNAL_ERROR',
      message: 'An error occurred while processing your search'
    });
  }
};
```

#### Service: `genie.service.js`

```javascript
// src/services/genie.service.js

const openai = require('../config/openai');
const posV2Api = require('./posV2Api.service');
const Restaurant = require('../models/Restaurant');
const logger = require('../utils/logger');

/**
 * Extract search intent from natural language query using GPT-4
 */
async function extractIntent(query) {
  const prompt = `You are a restaurant search assistant. Extract structured search intent from this query.

Query: "${query}"

Return JSON with:
- venueType: string (restaurant, bar, cafe, street_food, fine_dining, etc.)
- cuisine: string[] (lao, thai, western, japanese, korean, etc.)
- amenities: string[] (rooftop, outdoor_seating, wifi, parking, live_music, etc.)
- priceRange: string (budget, moderate, expensive)
- productPreference: string (beer, wine, cocktails, coffee, specific_dish_name)
- location: object with landmark/area name and approximate coordinates
- mealType: string (breakfast, lunch, dinner, snack)
- occasion: string (casual, date, business, family, celebration)

Only include fields that are explicitly mentioned or strongly implied in the query.`;

  try {
    const completion = await openai.chat.completions.create({
      model: 'gpt-4-turbo-preview',
      messages: [
        {
          role: 'system',
          content: 'You are a helpful assistant that extracts search intent from natural language restaurant queries. Return only valid JSON.'
        },
        {
          role: 'user',
          content: prompt
        }
      ],
      response_format: { type: 'json_object' },
      temperature: 0.3,
      max_tokens: 500
    });
    
    const intent = JSON.parse(completion.choices[0].message.content);
    
    logger.info('Extracted search intent', { query, intent });
    
    return intent;
    
  } catch (error) {
    logger.error('Failed to extract intent', { error: error.message });
    throw new Error('Failed to process search query');
  }
}

/**
 * Convert extracted intent to MongoDB query
 */
function buildMongoQuery(intent, userLocation) {
  const query = {};
  
  // Venue type
  if (intent.venueType) {
    query.tags = { $in: [intent.venueType] };
  }
  
  // Cuisine
  if (intent.cuisine && intent.cuisine.length > 0) {
    query.cuisine = { $in: intent.cuisine };
  }
  
  // Amenities
  if (intent.amenities && intent.amenities.length > 0) {
    query.amenities = { $all: intent.amenities };
  }
  
  // Price range
  if (intent.priceRange) {
    const priceMap = {
      budget: ['$', '$$'],
      moderate: ['$$', '$$$'],
      expensive: ['$$$', '$$$$']
    };
    query.priceRange = { $in: priceMap[intent.priceRange] || ['$$'] };
  }
  
  // Location proximity (if user location provided or intent has location)
  let searchLocation = userLocation;
  
  if (intent.location && intent.location.coordinates) {
    searchLocation = intent.location.coordinates;
  }
  
  if (searchLocation) {
    query.location = {
      $near: {
        $geometry: {
          type: 'Point',
          coordinates: [searchLocation.longitude, searchLocation.latitude]
        },
        $maxDistance: 5000 // 5km default
      }
    };
  }
  
  return query;
}

/**
 * Generate match score and explanation for each restaurant
 */
async function scoreRestaurant(restaurant, query, intent) {
  // Generate restaurant description for semantic matching
  const restaurantDescription = `
${restaurant.name} is a ${restaurant.cuisine.join(', ')} ${intent.venueType || 'restaurant'} 
with ${restaurant.priceRange} price range. 
Amenities: ${restaurant.amenities.join(', ')}.
Popular for: ${restaurant.popularDishes ? restaurant.popularDishes.join(', ') : 'various dishes'}.
${restaurant.highlights ? Object.entries(restaurant.highlights).map(([k, v]) => `${k}: ${v}`).join('. ') : ''}
  `.trim();
  
  try {
    // Get embeddings for both query and restaurant
    const embeddingResponse = await openai.embeddings.create({
      model: 'text-embedding-3-small',
      input: [query, restaurantDescription]
    });
    
    const [queryEmbedding, restaurantEmbedding] = embeddingResponse.data;
    
    // Calculate cosine similarity
    const similarity = cosineSimilarity(
      queryEmbedding.embedding,
      restaurantEmbedding.embedding
    );
    
    // Generate match reasons
    const matchReasons = [];
    
    if (intent.amenities) {
      const matchedAmenities = intent.amenities.filter(a => 
        restaurant.amenities.includes(a)
      );
      if (matchedAmenities.length > 0) {
        matchReasons.push(`Has ${matchedAmenities.join(' and ')}`);
      }
    }
    
    if (intent.priceRange === 'budget' && ['$', '$$'].includes(restaurant.priceRange)) {
      matchReasons.push(`Affordable pricing (${restaurant.priceRange})`);
    }
    
    if (restaurant.distance) {
      matchReasons.push(`Located ${restaurant.distance.toFixed(1)}km from you`);
    }
    
    if (intent.productPreference && restaurant.highlights) {
      const preference = intent.productPreference.toLowerCase();
      if (restaurant.highlights.cheapestBeerPrice && preference.includes('beer')) {
        matchReasons.push(`Beer prices under ${restaurant.highlights.cheapestBeerPrice.toLocaleString()} LAK`);
      }
    }
    
    // Generate AI explanation
    const explanationPrompt = `Explain in one sentence why "${restaurant.name}" is a good match for the query "${query}". Focus on the key match reasons: ${matchReasons.join(', ')}. Be conversational and helpful.`;
    
    const explanationResponse = await openai.chat.completions.create({
      model: 'gpt-4-turbo-preview',
      messages: [
        {
          role: 'user',
          content: explanationPrompt
        }
      ],
      max_tokens: 100,
      temperature: 0.7
    });
    
    const relevanceExplanation = explanationResponse.choices[0].message.content.trim();
    
    return {
      matchScore: similarity,
      matchReasons: matchReasons.slice(0, 3), // Top 3 reasons
      relevanceExplanation
    };
    
  } catch (error) {
    logger.error('Failed to score restaurant', { error: error.message });
    
    // Fallback to rule-based scoring if AI fails
    let fallbackScore = 0.5;
    
    if (intent.amenities) {
      const matchedCount = intent.amenities.filter(a => 
        restaurant.amenities.includes(a)
      ).length;
      fallbackScore += (matchedCount / intent.amenities.length) * 0.3;
    }
    
    return {
      matchScore: fallbackScore,
      matchReasons: ['Matches your search criteria'],
      relevanceExplanation: `${restaurant.name} matches your search for ${intent.venueType || 'a restaurant'}.`
    };
  }
}

/**
 * Calculate cosine similarity between two vectors
 */
function cosineSimilarity(vecA, vecB) {
  const dotProduct = vecA.reduce((sum, a, i) => sum + a * vecB[i], 0);
  const magnitudeA = Math.sqrt(vecA.reduce((sum, a) => sum + a * a, 0));
  const magnitudeB = Math.sqrt(vecB.reduce((sum, b) => sum + b * b, 0));
  return dotProduct / (magnitudeA * magnitudeB);
}

/**
 * Generate query refinement suggestions
 */
async function generateRefinements(query, intent, results) {
  const suggestions = [];
  
  // If price-sensitive, suggest different price ranges
  if (intent.priceRange === 'budget') {
    suggestions.push({
      suggestion: 'Show mid-range options',
      query: query.replace(/cheap|budget|affordable/gi, 'mid-range')
    });
  }
  
  // If location-specific, suggest expanding radius
  if (intent.location) {
    const areaName = intent.location.landmark || 'the area';
    suggestions.push({
      suggestion: 'Expand search area',
      query: query.replace(new RegExp(areaName, 'gi'), 'Vientiane')
    });
  }
  
  // If few results, suggest removing constraints
  if (results.length < 3) {
    suggestions.push({
      suggestion: 'Broaden search criteria',
      query: query.split(' ').slice(0, -2).join(' ') // Remove last 2 words
    });
  }
  
  // Suggest related amenities
  if (intent.amenities && intent.amenities.includes('rooftop')) {
    suggestions.push({
      suggestion: 'Show outdoor seating restaurants',
      query: query.replace(/rooftop/gi, 'outdoor seating')
    });
  }
  
  return suggestions.slice(0, 3); // Max 3 suggestions
}

/**
 * Main search function
 */
exports.search = async ({ query, userLocation, mode, maxResults, filters, userId }) => {
  const startTime = Date.now();
  
  try {
    // Step 1: Extract intent using AI
    const intent = await extractIntent(query);
    
    // Step 2: Build MongoDB query
    const mongoQuery = buildMongoQuery(intent, userLocation);
    
    // Step 3: Combine with hard filters
    Object.assign(mongoQuery, filters);
    
    // Step 4: Query restaurants from POS V2 + Consumer DB
    const restaurants = await posV2Api.searchRestaurants(mongoQuery);
    
    // Step 5: Score and rank restaurants using AI
    const scoredRestaurants = await Promise.all(
      restaurants.map(async (restaurant) => {
        const scoring = await scoreRestaurant(restaurant, query, intent);
        return {
          restaurantId: restaurant.id,
          name: restaurant.name,
          ...scoring,
          restaurant
        };
      })
    );
    
    // Step 6: Sort by match score, then distance, then rating
    scoredRestaurants.sort((a, b) => {
      // Primary: match score
      if (Math.abs(a.matchScore - b.matchScore) > 0.1) {
        return b.matchScore - a.matchScore;
      }
      // Secondary: distance (if available)
      if (a.restaurant.distance && b.restaurant.distance) {
        return a.restaurant.distance - b.restaurant.distance;
      }
      // Tertiary: rating
      return (b.restaurant.rating || 0) - (a.restaurant.rating || 0);
    });
    
    // Step 7: Limit results
    const results = scoredRestaurants.slice(0, maxResults);
    
    // Step 8: Generate refinement suggestions
    const refinementSuggestions = await generateRefinements(query, intent, results);
    
    const processingTime = Date.now() - startTime;
    
    return {
      query,
      queryIntent: intent,
      results,
      refinementSuggestions,
      totalResults: results.length,
      searchMetadata: {
        searchId: `search_${userId}_${Date.now()}`,
        processingTime,
        aiModel: 'gpt-4-turbo',
        confidenceScore: results.length > 0 ? results[0].matchScore : 0
      }
    };
    
  } catch (error) {
    logger.error('AI search failed', { error: error.message, query });
    throw error;
  }
};
```

#### OpenAI API Costs (December 2025 pricing)

| Operation | Model | Cost per 1K tokens | Tokens per Request | Cost per Search |
|-----------|-------|-------------------|-------------------|-----------------|
| Intent extraction | GPT-4 Turbo | $0.01 input / $0.03 output | ~500 input, 200 output | $0.011 |
| Embeddings | text-embedding-3-small | $0.00002 | ~1000 tokens | $0.00002 |
| Match explanation | GPT-4 Turbo | $0.01 input / $0.03 output | ~200 input, 50 output | $0.0035 |
| **Total per search** | | | | **~$0.015** |

**Monthly Cost Projections**:

| Scenario | Daily Searches | Monthly Cost | Notes |
|----------|---------------|--------------|-------|
| Soft Launch | 100 searches/day | $45/month | Testing phase |
| Growing | 500 searches/day | $225/month | Active user base |
| Scale | 2000 searches/day | $900/month | Mature product |

**Cost Optimization Strategies**:

1. **Cache intent extraction** for identical queries (30-day TTL)
2. **Use Claude API** as fallback (cheaper for some operations)
3. **Implement query templates** for common patterns
4. **Batch embeddings** for multiple restaurants
5. **Rate limit** aggressively (30 req/min per user)

## Feature 2: Bulk Voucher Campaigns

### Product Requirements

**User Story:**
> As a Marketing Manager, I want to create bulk voucher campaigns (e.g., "Grand Slam: Pay 75k, Get 100k") so that I can run promotional sales and pre-sell credits to generate cash flow.

**Key Features:**
1. Bulk voucher generation (up to 10,000 codes per campaign)
2. Campaign management (create, pause, activate, analytics)
3. Customizable voucher parameters (value, validity, conditions)
4. CSV export of voucher codes for distribution
5. Real-time usage tracking and analytics
6. Automated notifications when thresholds hit

### Implementation Code

#### Controller: `voucherCampaign.controller.js`

```javascript
// src/controllers/voucherCampaign.controller.js

const VoucherCampaign = require('../models/VoucherCampaign');
const Voucher = require('../models/Voucher');
const { ValidationError, ForbiddenError } = require('../utils/errors');
const logger = require('../utils/logger');
const crypto = require('crypto');
const { Parser } = require('json2csv');

/**
 * Generate unique voucher code
 */
function generateVoucherCode(prefix, index) {
  const paddedIndex = index.toString().padStart(5, '0');
  return `${prefix}-${paddedIndex}`;
}

/**
 * Create bulk voucher campaign
 */
exports.createCampaign = async (req, res) => {
  try {
    const adminId = req.user.id;
    
    // Check admin permissions
    if (!req.user.roles.includes('admin')) {
      throw new ForbiddenError('Admin role required');
    }
    
    const {
      campaignName,
      campaignCode,
      description,
      quantity,
      pricing,
      validity,
      conditions,
      distribution,
      autoActivate = false
    } = req.body;
    
    // Validate quantity
    if (quantity < 1 || quantity > 10000) {
      throw new ValidationError('Quantity must be between 1 and 10,000');
    }
    
    // Validate pricing
    if (pricing.creditValue <= pricing.purchasePrice) {
      throw new ValidationError('Credit value must be greater than purchase price');
    }
    
    // Check if campaign code already exists
    const existingCampaign = await VoucherCampaign.findOne({ campaignCode });
    if (existingCampaign) {
      return res.status(409).json({
        error: 'DUPLICATE_CAMPAIGN_CODE',
        message: `Campaign code '${campaignCode}' already exists`,
        existingCampaignId: existingCampaign._id
      });
    }
    
    // Calculate discount
    const discount = pricing.creditValue - pricing.purchasePrice;
    const discountPercent = (discount / pricing.creditValue) * 100;
    
    // Calculate valid days
    const startDate = new Date(validity.startDate);
    const endDate = new Date(validity.endDate);
    const validDays = Math.ceil((endDate - startDate) / (1000 * 60 * 60 * 24));
    
    // Create campaign
    const campaign = await VoucherCampaign.create({
      campaignName,
      campaignCode,
      description,
      quantity,
      pricing: {
        ...pricing,
        discount,
        discountPercent
      },
      validity: {
        startDate,
        endDate,
        validDays
      },
      conditions: conditions || {},
      distribution: distribution || { method: 'manual' },
      status: autoActivate ? 'active' : 'draft',
      createdBy: adminId,
      stats: {
        total: quantity,
        purchased: 0,
        redeemed: 0,
        expired: 0,
        totalRevenue: 0,
        totalDiscount: 0
      }
    });
    
    // Generate voucher codes in batches
    logger.info(`Generating ${quantity} vouchers for campaign ${campaign._id}`);
    
    const batchSize = 100;
    const vouchers = [];
    
    for (let i = 0; i < quantity; i++) {
      const voucherCode = generateVoucherCode(
        campaignCode.replace(/_/g, '-'),
        i + 1
      );
      
      vouchers.push({
        code: voucherCode,
        campaignId: campaign._id,
        purchasePrice: pricing.purchasePrice,
        creditValue: pricing.creditValue,
        discount,
        type: 'credit',
        status: 'available',
        validFrom: startDate,
        validUntil: endDate,
        usageLimit: conditions?.usageLimit || 1,
        usageCount: 0,
        conditions: {
          minOrderAmount: conditions?.minOrderAmount || 0,
          maxDiscountPerOrder: conditions?.maxDiscountPerOrder || pricing.creditValue,
          applicableProducts: conditions?.applicableProducts || ['eats', 'live', 'market']
        }
      });
      
      // Insert in batches for performance
      if (vouchers.length >= batchSize || i === quantity - 1) {
        await Voucher.insertMany(vouchers);
        vouchers.length = 0; // Clear array
      }
    }
    
    logger.info(`Successfully generated ${quantity} vouchers for campaign ${campaign._id}`);
    
    // Sample codes for response
    const sampleCodes = [];
    for (let i = 0; i < Math.min(3, quantity); i++) {
      sampleCodes.push(generateVoucherCode(campaignCode.replace(/_/g, '-'), i + 1));
    }
    if (quantity > 3) {
      sampleCodes.push(`...${quantity - 3} more`);
    }
    
    // Return response
    return res.status(201).json({
      campaignId: campaign._id,
      campaignName: campaign.campaignName,
      campaignCode: campaign.campaignCode,
      status: campaign.status,
      quantity: campaign.quantity,
      pricing: campaign.pricing,
      validity: campaign.validity,
      vouchersGenerated: quantity,
      voucherCodesSample: sampleCodes,
      downloadUrl: `${process.env.CONSUMER_API_URL}/v1/admin/vouchers/campaigns/${campaign._id}/export`,
      stats: campaign.stats,
      createdAt: campaign.createdAt,
      createdBy: {
        userId: adminId,
        name: req.user.fullName
      }
    });
    
  } catch (error) {
    logger.error('Failed to create voucher campaign', { error: error.message });
    
    if (error.name === 'ValidationError') {
      return res.status(400).json({
        error: 'VALIDATION_ERROR',
        message: error.message
      });
    }
    
    if (error.name === 'ForbiddenError') {
      return res.status(403).json({
        error: 'INSUFFICIENT_PERMISSIONS',
        message: error.message
      });
    }
    
    return res.status(500).json({
      error: 'INTERNAL_ERROR',
      message: 'Failed to create voucher campaign'
    });
  }
};

exports.listCampaigns = async (req, res) => {
  try {
    const { status, limit = 20, offset = 0 } = req.query;
    
    const query = {};
    if (status) {
      query.status = status;
    }
    
    const campaigns = await VoucherCampaign.find(query)
      .select('campaignId campaignName campaignCode status quantity pricing validity stats createdAt')
      .sort({ createdAt: -1 })
      .limit(parseInt(limit))
      .skip(parseInt(offset));
    
    const total = await VoucherCampaign.countDocuments(query);
    
    return res.json({
      campaigns,
      pagination: {
        total,
        limit: parseInt(limit),
        offset: parseInt(offset),
        hasMore: offset + campaigns.length < total
      }
    });
    
  } catch (error) {
    logger.error('Failed to list campaigns', { error: error.message });
    return res.status(500).json({
      error: 'INTERNAL_ERROR',
      message: 'Failed to list campaigns'
    });
  }
};

/**
 * Get campaign details
 */
exports.getCampaign = async (req, res) => {
  try {
    const { campaignId } = req.params;
    
    const campaign = await VoucherCampaign.findById(campaignId)
      .populate('createdBy', 'fullName email');
    
    if (!campaign) {
      return res.status(404).json({
        error: 'CAMPAIGN_NOT_FOUND',
        message: 'Campaign not found'
      });
    }
    
    // Calculate days remaining
    const now = new Date();
    const endDate = new Date(campaign.validity.endDate);
    const daysRemaining = Math.max(0, Math.ceil((endDate - now) / (1000 * 60 * 60 * 24)));
    
    // Get top performing channels
    const channelStats = await Voucher.aggregate([
      { $match: { campaignId: campaign._id, status: 'purchased' } },
      { $group: {
        _id: '$purchaseChannel',
        purchases: { $sum: 1 },
        revenue: { $sum: '$purchasePrice' }
      }},
      { $sort: { purchases: -1 } },
      { $limit: 3 }
    ]);
    
    return res.json({
      campaignId: campaign._id,
      campaignName: campaign.campaignName,
      campaignCode: campaign.campaignCode,
      status: campaign.status,
      quantity: campaign.quantity,
      pricing: campaign.pricing,
      validity: {
        ...campaign.validity.toObject(),
        daysRemaining
      },
      conditions: campaign.conditions,
      stats: {
        ...campaign.stats,
        purchaseRate: (campaign.stats.purchased / campaign.stats.total * 100).toFixed(1),
        redemptionRate: campaign.stats.purchased > 0
          ? (campaign.stats.redeemed / campaign.stats.purchased * 100).toFixed(1)
          : 0
      },
      timeline: {
        createdAt: campaign.createdAt,
        activatedAt: campaign.activatedAt,
        firstPurchase: campaign.firstPurchaseAt,
        firstRedemption: campaign.firstRedemptionAt
      },
      topPerformingChannels: channelStats.map(s => ({
        channel: s._id,
        purchases: s.purchases,
        revenue: s.revenue
      }))
    });
    
  } catch (error) {
    logger.error('Failed to get campaign', { error: error.message, campaignId: req.params.campaignId });
    return res.status(500).json({
      error: 'INTERNAL_ERROR',
      message: 'Failed to get campaign details'
    });
  }
};

exports.exportCampaign = async (req, res) => {
  try {
    const { campaignId } = req.params;
    
    const campaign = await VoucherCampaign.findById(campaignId);
    if (!campaign) {
      return res.status(404).json({
        error: 'CAMPAIGN_NOT_FOUND',
        message: 'Campaign not found'
      });
    }
    
    // Get all vouchers for this campaign
    const vouchers = await Voucher.find({ campaignId })
      .populate('purchasedBy', 'phone fullName')
      .sort({ code: 1 });
    
    // Convert to CSV format
    const csvData = vouchers.map(v => ({
      'Voucher Code': v.code,
      'Purchase Price': v.purchasePrice,
      'Credit Value': v.creditValue,
      'Valid From': v.validFrom.toISOString().split('T')[0],
      'Valid Until': v.validUntil.toISOString().split('T')[0],
      'Status': v.status,
      'Purchased By': v.purchasedBy ? v.purchasedBy.phone : '',
      'Purchased At': v.purchasedAt ? v.purchasedAt.toISOString() : '',
      'Redeemed At': v.redeemedAt ? v.redeemedAt.toISOString() : ''
    }));
    
    const parser = new Parser();
    const csv = parser.parse(csvData);
    
    const filename = `${campaign.campaignCode.toLowerCase()}_vouchers_${Date.now()}.csv`;
    
    res.setHeader('Content-Type', 'text/csv');
    res.setHeader('Content-Disposition', `attachment; filename="${filename}"`);
    res.send(csv);
    
  } catch (error) {
    logger.error('Failed to export campaign', { error: error.message, campaignId: req.params.campaignId });
    return res.status(500).json({
      error: 'INTERNAL_ERROR',
      message: 'Failed to export campaign'
    });
  }
};

/**
 * Update campaign status
 */
exports.updateCampaign = async (req, res) => {
  try {
    const { campaignId } = req.params;
    const { status, validity } = req.body;
    
    const campaign = await VoucherCampaign.findById(campaignId);
    if (!campaign) {
      return res.status(404).json({
        error: 'CAMPAIGN_NOT_FOUND',
        message: 'Campaign not found'
      });
    }
    
    // Update status
    if (status) {
      // Validate status transition
      const validTransitions = {
        draft: ['active'],
        active: ['paused'],
        paused: ['active']
      };
      
      if (!validTransitions[campaign.status]?.includes(status)) {
        return res.status(400).json({
          error: 'INVALID_STATUS_TRANSITION',
          message: `Cannot change status from ${campaign.status} to ${status}`
        });
      }
      
      campaign.status = status;
      
      if (status === 'active' && !campaign.activatedAt) {
        campaign.activatedAt = new Date();
      }
    }
    
    // Update validity
    if (validity?.endDate) {
      const newEndDate = new Date(validity.endDate);
      const now = new Date();
      
      if (newEndDate < now) {
        return res.status(400).json({
          error: 'INVALID_END_DATE',
          message: 'End date cannot be in the past'
        });
      }
      
      campaign.validity.endDate = newEndDate;
      campaign.validity.validDays = Math.ceil(
        (newEndDate - campaign.validity.startDate) / (1000 * 60 * 60 * 24)
      );
      
      // Update all voucher validUntil dates
      await Voucher.updateMany(
        { campaignId: campaign._id },
        { validUntil: newEndDate }
      );
    }
    
    await campaign.save();
    
    return res.json({
      campaignId: campaign._id,
      status: campaign.status,
      message: `Campaign successfully ${status === 'active' ? 'activated' : status === 'paused' ? 'paused' : 'updated'}`
    });
    
  } catch (error) {
    logger.error('Failed to update campaign', { error: error.message, campaignId: req.params.campaignId });
    return res.status(500).json({
      error: 'INTERNAL_ERROR',
      message: 'Failed to update campaign'
    });
  }
};
```
