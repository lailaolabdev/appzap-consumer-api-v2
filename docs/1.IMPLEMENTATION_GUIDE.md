### Step 4: Testing Strategy

#### Unit Tests (80% coverage target)

```javascript
// tests/unit/loyalty.test.js
describe('Loyalty Points Calculation', () => {
  it('should calculate earned points correctly', () => {
    const orderTotal = 150000; // 150,000 LAK
    const points = calculateEarnedPoints(orderTotal);
    expect(points).toBe(150); // 150,000 / 1000 = 150 points
  });
  
  it('should calculate redeemed discount correctly', () => {
    const pointsToRedeem = 100;
    const discount = calculateLoyaltyDiscount(pointsToRedeem);
    expect(discount).toBe(5000); // 100 * 50 = 5,000 LAK
  });
});
```

#### Integration Tests

```javascript
// tests/integration/auth.test.js
describe('Authentication Flow', () => {
  it('should complete OTP login flow', async () => {
    // Request OTP
    const otpResponse = await request(app)
      .post('/v1/auth/request-otp')
      .send({ phone: '8562093352677' })
      .expect(200);
    
    // Verify OTP (use test OTP in test environment)
    const loginResponse = await request(app)
      .post('/v1/auth/verify-otp')
      .send({ phone: '8562093352677', otp: '123456' })
      .expect(200);
    
    expect(loginResponse.body.accessToken).toBeDefined();
    expect(loginResponse.body.user).toBeDefined();
  });
});
```

#### E2E Tests

```javascript
// tests/e2e/orderFlow.test.js
describe('Complete Order Flow', () => {
  it('should create and pay for order', async () => {
    // Login
    const auth = await loginUser();
    
    // Create cart
    const cart = await createCart(auth.token);
    
    // Add items
    await addItemToCart(cart.id, { menuItemId: 'item_001', quantity: 2 });
    
    // Checkout
    const order = await checkoutCart(cart.id);
    
    // Verify order created
    expect(order.orderId).toBeDefined();
    expect(order.paymentUrl).toBeDefined();
  });
});
```
