### Database Name

- **Development**: `appzap_consumer_dev`
- **Staging**: `appzap_consumer_staging`
- **Production**: `appzap_consumer_prod`

```javascript
UserSchema.virtual('activeProfileDetails').get(function() {

});

// Methods
UserSchema.methods.toJSON = function() {
  const user = this.toObject();
  delete user.__v;
  delete user.isDeleted;
  return user;
};

module.exports = mongoose.model('User', UserSchema);
```

```javascript
    type: Number,

const CartSchema = new Schema({
  userId: {
    type: Schema.Types.ObjectId,
    ref: 'User',
    required: true,
    index: true
  },
  restaurantId: {
    type: String,
    required: true,
    trim: true,
    index: true
  },
  restaurantName: {
    type: String,
    trim: true
  },
  orderType: {
    type: String,
    enum: ['dine_in', 'takeaway'],
    required: true
  },
  tableId: {
    type: String,
    trim: true
  },
  items: [CartItemSchema],
  
  // Pricing
  subtotal: {
    type: Number,
    default: 0,
    min: 0
  },
  discount: {
    type: Number,
    default: 0,
    min: 0
  },
  appliedVoucher: {
    voucherId: Schema.Types.ObjectId,
    code: String,
    discountAmount: Number
  },
  total: {
    type: Number,
    default: 0,
    min: 0
  },
  
  // Deep Link Attribution
  deepLinkData: {
    source: String,
    referralCode: String,
    campaignId: String
  },
  
  // Expiration
  expiresAt: {
    type: Date,
    required: true,
    index: true,
    default: () => new Date(Date.now() + 60 * 60 * 1000) // 60 minutes
  }
  
}, {
  timestamps: true,
  collection: 'carts'
});

// Indexes
CartSchema.index({ userId: 1, createdAt: -1 });
CartSchema.index({ expiresAt: 1 }, { expireAfterSeconds: 0 }); // TTL index
CartSchema.index({ restaurantId: 1 });

// Pre-save middleware to calculate totals
CartSchema.pre('save', function(next) {
  this.subtotal = this.items.reduce((sum, item) => sum + item.itemTotal, 0);
  this.total = this.subtotal - this.discount;
  next();
});

module.exports = mongoose.model('Cart', CartSchema);
```

```javascript
function generateOrderCode(type = 'eats') {
  const prefix = type === 'market' ? 'MKT' : 'AZ';
  const date = new Date();
  const dateStr = date.toISOString().split('T')[0].replace(/-/g, '');
  const random = Math.floor(Math.random() * 1000).toString().padStart(3, '0');
  return `${prefix}-${dateStr}-${random}`;
}
```
