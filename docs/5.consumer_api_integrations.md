## Integration Overview

### External Systems

The Consumer API integrates with 4 external systems:

| System | Purpose | Protocol | Authentication |
|--------|---------|----------|----------------|
| **Auth API** | OTP authentication | GraphQL over HTTPS | API Key (platform) |
| **POS V2 API** | Restaurant orders, menus | REST + WebSocket | API Key or JWT |
| **Supplier API** | Market products, orders | REST | Exchange Key |
| **Phapay** | Payment processing | REST + Webhooks | Merchant credentials |

### HTTP Client Configuration

```javascript
// config/httpClient.js
const axios = require('axios');
const axiosRetry = require('axios-retry');

// Base HTTP client with common settings
const createHttpClient = (baseURL, options = {}) => {
  const client = axios.create({
    baseURL,
    timeout: options.timeout || 30000, // 30 seconds
    headers: {
      'Content-Type': 'application/json',
      'User-Agent': 'AppZap-Consumer-API/1.0.0',
      ...options.headers
    }
  });
  
  // Request interceptor
  client.interceptors.request.use(
    (config) => {
      // Add request ID for tracing
      config.headers['X-Request-ID'] = uuidv4();
      
      // Log request
      logger.debug('External API request', {
        method: config.method,
        url: config.url,
        baseURL: config.baseURL
      });
      
      return config;
    },
    (error) => {
      logger.error('Request interceptor error', { error: error.message });
      return Promise.reject(error);
    }
  );
  
  // Response interceptor
  client.interceptors.response.use(
    (response) => {
      // Log successful response
      logger.debug('External API response', {
        status: response.status,
        url: response.config.url,
        duration: response.headers['x-response-time']
      });
      
      return response;
    },
    (error) => {
      // Log error response
      if (error.response) {
        logger.error('External API error response', {
          status: error.response.status,
          url: error.config?.url,
          error: error.response.data
        });
      } else if (error.request) {
        logger.error('External API no response', {
          url: error.config?.url,
          error: error.message
        });
      }
      
      return Promise.reject(error);
    }
  );
  
  // Configure retries
  if (options.retry !== false) {
    axiosRetry(client, {
      retries: options.retries || 3,
      retryDelay: axiosRetry.exponentialDelay,
      retryCondition: (error) => {
        // Retry on network errors or 5xx status codes
        return axiosRetry.isNetworkOrIdempotentRequestError(error) ||
               (error.response?.status >= 500 && error.response?.status < 600) ||
               error.response?.status === 429; // Rate limit
      },
      onRetry: (retryCount, error, requestConfig) => {
        logger.warn('Retrying external API request', {
          retryCount,
          url: requestConfig.url,
          error: error.message
        });
      }
    });
  }
  
  return client;
};

module.exports = { createHttpClient };
```

const REQUEST_OTP_MUTATION = `
  mutation RequestOtpMutation($data: OtpInput!) {
    requestOtp(data: $data) {
      message
    }
  }
`;

exports.requestOTP = async ({ phone, platform = 'APPZAP', header = 'AppZap' }) => {
  try {

      }

  }

### Verify OTP & Login

```javascript
const PHONE_LOGIN_MUTATION = `
  mutation PhoneLogin($where: PhoneLoginInput!) {
    phoneLogin(where: $where) {
      accessToken
      refreshToken
      data {
        id
        phone
        nickName
        role
        createdAt
        updatedAt
      }
    }
  }
`;

exports.verifyOTP = async ({ phone, otp }) => {
  try {
    const variables = {
      where: {
        phone,
        otp
      }
    };
    
    const response = await authApiClient.request(PHONE_LOGIN_MUTATION, variables);
    
    if (!response.phoneLogin) {
      logger.warn('OTP verification failed', { phone: maskPhone(phone) });
      return {
        success: false,
        error: 'Invalid or expired OTP'
      };
    }
    
    logger.info('OTP verified successfully', { 
      phone: maskPhone(phone),
      userId: response.phoneLogin.data.id
    });
    
    return {
      success: true,
      user: response.phoneLogin.data,
      authApiTokens: {
        accessToken: response.phoneLogin.accessToken,
        refreshToken: response.phoneLogin.refreshToken
      }
    };
    
  } catch (error) {
    logger.error('Auth API verifyOTP failed', {
      error: error.message,
      phone: maskPhone(phone)
    });
    
    return {
      success: false,
      error: error.response?.errors?.[0]?.message || 'OTP verification failed'
    };
  }
};

### Error Handling

// Auth API error codes mapping
const AUTH_API_ERRORS = {
  'INVALID_OTP': 'Invalid or expired OTP',
  'OTP_EXPIRED': 'OTP has expired. Please request a new one.',
  'PHONE_NOT_FOUND': 'Phone number not registered',
  'TOO_MANY_REQUESTS': 'Too many OTP requests. Please try again later.'

exports.handleAuthApiError = (error) => {
  const graphqlError = error.response?.errors?.[0];
  
  if (!graphqlError) {
    return {
      code: 'AUTH_API_ERROR',
      message: 'Authentication service is unavailable',
      status: 503
    };
  }
  
  const errorCode = graphqlError.extensions?.code;
  
  return {
    code: errorCode || 'AUTH_API_ERROR',
    message: AUTH_API_ERRORS[errorCode] || graphqlError.message,
    status: 400
  };
};
```

### WebSocket Integration (Live Bills)

```javascript
// services/posV2Websocket.service.js
const io = require('socket.io-client');

class PosV2WebSocketService {
  constructor() {
    this.socket = null;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = 5;
  }
  
  connect() {
    const POS_WS_URL = process.env.POS_V2_WS_URL || 'ws://localhost:8080';
    
    this.socket = io(POS_WS_URL, {
      auth: {
        apiKey: process.env.POS_V2_API_KEY
      },
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionDelayMax: 5000,
      reconnectionAttempts: this.maxReconnectAttempts
    });
    
    this.socket.on('connect', () => {
      logger.info('Connected to POS V2 WebSocket');
      this.reconnectAttempts = 0;
    });
    
    this.socket.on('disconnect', (reason) => {
      logger.warn('Disconnected from POS V2 WebSocket', { reason });
    });
    
    this.socket.on('connect_error', (error) => {
      this.reconnectAttempts++;
      logger.error('POS V2 WebSocket connection error', {
        error: error.message,
        attempt: this.reconnectAttempts
      });
      
      if (this.reconnectAttempts >= this.maxReconnectAttempts) {
        logger.error('Max reconnect attempts reached for POS V2 WebSocket');
      }
    });
  }
  
  subscribeToBill(billId, callback) {
    if (!this.socket) {
      this.connect();
    }
    
    // Join bill room
    this.socket.emit('join_bill', billId);
    
    // Listen for bill updates
    this.socket.on(`bill_updated:${billId}`, (data) => {
      logger.debug('Bill update received', { billId });
      callback(data);
    });
    
    logger.info('Subscribed to bill updates', { billId });
  }
  
  unsubscribeFromBill(billId) {
    if (this.socket) {
      this.socket.emit('leave_bill', billId);
      this.socket.off(`bill_updated:${billId}`);
      logger.info('Unsubscribed from bill updates', { billId });
    }
  }
  
  disconnect() {
    if (this.socket) {
      this.socket.disconnect();
      this.socket = null;
    }
  }
}

module.exports = new PosV2WebSocketService();
```

### Authentication

```javascript
exports.authenticate = async (phone) => {
  try {
    // Check cache first (24h TTL)
    const cacheKey = `supplier:token:${phone}`;
    const cachedToken = await redis.get(cacheKey);
    
    if (cachedToken) {
      logger.debug('Supplier token served from cache', { phone: maskPhone(phone) });
      return cachedToken;
    }
    
    // Normalize phone number for Supplier API
    const normalizedPhone = normalizePhoneForSupplier(phone);
    
    // Authenticate with Supplier API
    const response = await supplierClient.post('/user/authenticate/token', {
      phone: normalizedPhone,
      exchangeKey: SUPPLIER_EXCHANGE_KEY
    });
    
    const token = response.data.accessToken;
    const customerId = response.data.customerId;
    
    // Cache token for 24 hours
    await redis.setex(cacheKey, 86400, token);
    
    // Store customer ID mapping
    await redis.setex(`supplier:customerId:${phone}`, 86400, customerId);
    
    logger.info('Authenticated with Supplier API', {
      phone: maskPhone(phone),
      customerId
    });
    
    return token;
    
  } catch (error) {
    logger.error('Supplier API authentication failed', {
      phone: maskPhone(phone),
      error: error.message,
      status: error.response?.status
    });
    
    throw new Error('Failed to authenticate with Supplier API');
  }
};

function normalizePhoneForSupplier(phone) {
  // Remove leading 0 or 856
  return phone.replace(/^(0|856)/, '');
}
```

### Get Products

```javascript
exports.getProducts = async ({ phone, categoryId, skip = 0, limit = 24 }) => {
  try {
    // Authenticate
    const token = await this.authenticate(phone);
    
    // Check cache
    const cacheKey = `supplier:products:${categoryId}:${skip}:${limit}`;
    const cached = await redis.get(cacheKey);
    
    if (cached) {
      return {
        success: true,
        data: JSON.parse(cached),
        cached: true
      };
    }
    
    // Fetch from Supplier API
    const response = await supplierClient.get(
      `/mobile/product/get/skip/${skip}/limit/${limit}`,
      {
        headers: {
          Authorization: `Bearer ${token}`
        },
        params: {
          categoryId,
          createdBy: SUPPLIER_SELLER_ID
        }
      }
    );
    
    // Cache for 10 minutes
    await redis.setex(cacheKey, 600, JSON.stringify(response.data));
    
    return {
      success: true,
      data: response.data.data,
      total: response.data.total,
      cached: false
    };
    
  } catch (error) {
    logger.error('Supplier API getProducts failed', {
      categoryId,
      error: error.message,
      status: error.response?.status
    });
    
    // Handle token expiration
    if (error.response?.status === 401) {
      // Clear cached token and retry
      await redis.del(`supplier:token:${phone}`);
      logger.warn('Supplier token expired, retrying with fresh token');
      
      // Retry once
      return this.getProducts({ phone, categoryId, skip, limit });
    }
    
    throw error;
  }
};
```

      paymentId: response.data.paymentId,
      amount: paymentData.amount,
      status: 'pending',
      paymentUrl: response.data.paymentUrl,
      qrCode: response.data.qrCode
    });
    
    logger.info('Phapay payment session created', {
      orderId: paymentData.orderId,
      paymentId: response.data.paymentId,
      amount: paymentData.amount
    });
    
    return {
      success: true,
      paymentId: response.data.paymentId,
      paymentUrl: response.data.paymentUrl,
      qrCode: response.data.qrCode,
      expiresAt: new Date(Date.now() + 15 * 60000) // 15 minutes
    };
    
  } catch (error) {
    logger.error('Phapay createPaymentSession failed', {
      orderId: paymentData.orderId,
      error: error.message,
      status: error.response?.status
    });
    
    throw new Error('Failed to create payment session');
  }
};

function generatePhapaySignature(payload) {
  const sortedKeys = Object.keys(payload).sort();
  const signString = sortedKeys.map(key => `${key}=${payload[key]}`).join('&');
  
  return crypto
    .createHmac('sha256', PHAPAY_SECRET_KEY)
    .update(signString)
    .digest('hex');
}

### Exponential Backoff Retry

```javascript
// utils/retry.js
exports.retryWithBackoff = async (fn, options = {}) => {
  const {
    maxRetries = 3,
    initialDelay = 1000,
    maxDelay = 30000,
    backoffFactor = 2,
    onRetry = () => {}
  } = options;
  
  let lastError;
  
  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error;
      
      if (attempt < maxRetries) {
        const delay = Math.min(
          initialDelay * Math.pow(backoffFactor, attempt),
          maxDelay
        );
        
        onRetry(attempt + 1, error, delay);
        
        await new Promise(resolve => setTimeout(resolve, delay));
      }
    }
  }
  
  throw lastError;
};

// Usage:
const result = await retryWithBackoff(
  () => posV2Api.createOrder(orderData),
  {
    maxRetries: 3,
    initialDelay: 1000,
    onRetry: (attempt, error, delay) => {
      logger.warn('Retrying POS order creation', {
        attempt,
        delay,
        error: error.message
      });
    }
  }
);
```

// jobs/retryFailedSyncs.js
const Queue = require('bull');
const Order = require('../models/Order');
const MarketOrder = require('../models/MarketOrder');
const posV2Api = require('../services/posV2Api.service');
const supplierApi = require('../services/supplierApi.service');

// Process retry job
retryQueue.process(async (job) => {
  const { type, orderId } = job.data;
  
  if (type === 'pos_order') {
    const order = await Order.findById(orderId);
    
    if (!order || order.posSyncStatus === 'synced') {
      return { skipped: true };
    }
    
    // Retry POS sync
    const result = await posV2Api.createOrder({
      consumerOrderId: order._id,
      restaurantId: order.restaurantId,
      items: order.items,
      customerName: order.customerName,
      customerPhone: order.customerPhone
    });
    
    if (result.success) {
      order.posOrderId = result.posOrderId;
      order.posSyncStatus = 'synced';
      order.lastPosSyncAt = new Date();
      await order.save();
      
      logger.info('POS sync retry succeeded', { orderId: order._id });
      return { success: true };
    } else {
      order.posSyncRetries += 1;
      order.posSyncError = result.error;
      await order.save();
      
      // Schedule another retry if under max attempts
      if (order.posSyncRetries < 5) {
        await retryQueue.add(
          { type: 'pos_order', orderId: order._id },
          { delay: 60000 * order.posSyncRetries } // Exponential delay
        );
      }
      
      throw new Error(result.error);
    }
  }
  
  // Similar logic for supplier orders...
});

// Schedule retry for failed syncs (runs every 5 minutes)
const cron = require('node-cron');

cron.schedule('*/5 * * * *', async () => {
  // Find orders with failed syncs

    posSyncRetries: { $lt: 5 },
    createdAt: { $gte: new Date(Date.now() - 24 * 60 * 60 * 1000) } // Last 24h only
  });
  
  for (const order of failedPosOrders) {
    await retryQueue.add({ type: 'pos_order', orderId: order._id });
  }
  
  logger.info('Scheduled retry jobs', { count: failedPosOrders.length });
});

// utils/circuitBreaker.js
const CircuitBreaker = require('opossum');

exports.createCircuitBreaker = (fn, options = {}) => {
  const breaker = new CircuitBreaker(fn, {
    timeout: options.timeout || 30000,
    errorThresholdPercentage: 50,
    resetTimeout: 30000,
    ...options
  });
  
  breaker.on('open', () => {
    logger.warn('Circuit breaker opened', { name: options.name });
  });
  
  breaker.on('halfOpen', () => {
    logger.info('Circuit breaker half-open', { name: options.name });
  });
  
  breaker.on('close', () => {
    logger.info('Circuit breaker closed', { name: options.name });
  });
  
  breaker.fallback(() => {
    if (options.fallback) {
      return options.fallback();
    }
    throw new Error('Service temporarily unavailable');
  });
  
  return breaker;
};

// Usage:
const posOrderBreaker = createCircuitBreaker(
  (orderData) => posV2Api.createOrder(orderData),
  {
    name: 'pos-create-order',
    timeout: 30000,
    fallback: () => ({
      success: false,
      error: 'POS service temporarily unavailable',
      code: 'SERVICE_UNAVAILABLE'
    })
  }
);

// Use circuit breaker
const result = await posOrderBreaker.fire(orderData);
