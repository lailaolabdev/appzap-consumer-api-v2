## Authentication System

### Overview

The Consumer API uses **OTP-based authentication** with JWT tokens for session management.

**Flow**:
```
User → Request OTP → Auth API verifies → Consumer API issues JWT → User authenticated
```

---

### OTP Authentication Implementation

#### 1. Request OTP Endpoint

```javascript
// controllers/auth.controller.js
const redis = require('../config/redis');
const authApiClient = require('../services/authApi.service');

exports.requestOTP = async (req, res) => {
  try {
    const { phone } = req.body;
    
    // Validate phone format
    if (!validateLaoPhone(phone)) {
      return res.status(400).json({
        error: {
          code: 'INVALID_PHONE',
          message: 'Invalid Lao phone number format'
        }
      });
    }
    
    // Rate limiting: 3 requests per 5 minutes per phone
    const rateLimitKey = `otp:rate:${phone}`;
    const attempts = await redis.incr(rateLimitKey);
    
    if (attempts === 1) {
      await redis.expire(rateLimitKey, 300); // 5 minutes
    }
    
    if (attempts > 3) {
      return res.status(429).json({
        error: {
          code: 'RATE_LIMIT_EXCEEDED',
          message: 'Too many OTP requests. Please try again in 5 minutes.',
          retryAfter: 300
        }
      });
    }
    
    // Forward to Auth API
    const result = await authApiClient.requestOTP({
      phone,
      platform: 'APPZAP',
      header: 'AppZap'
    });
    
    // Log for security monitoring
    logger.info('OTP requested', {
      phone: maskPhone(phone),
      ip: req.ip,
      userAgent: req.get('user-agent')
    });
    
    return res.json({
      message: 'OTP sent successfully',
      referenceId: result.referenceId,
      expiresIn: 300
    });
    
  } catch (error) {
    logger.error('OTP request failed', { error: error.message, phone: maskPhone(req.body.phone) });
    return res.status(500).json({
      error: {
        code: 'OTP_REQUEST_FAILED',
        message: 'Failed to send OTP. Please try again.'
      }
    });
  }
};

function validateLaoPhone(phone) {
  return /^(856)?\d{8,10}$/.test(phone);
}

function maskPhone(phone) {
  return phone.replace(/(\d{3})\d+(\d{4})/, '$1****$2');
}
```

#### 2. Verify OTP & Login Endpoint

```javascript
exports.verifyOTP = async (req, res) => {
  try {
    const { phone, otp } = req.body;
    
    // Validate inputs
    if (!phone || !otp) {
      return res.status(400).json({
        error: {
          code: 'MISSING_FIELDS',
          message: 'Phone and OTP are required'
        }
      });
    }
    
    // Rate limiting: 5 attempts per 10 minutes per phone
    const verifyRateLimitKey = `otp:verify:${phone}`;
    const verifyAttempts = await redis.incr(verifyRateLimitKey);
    
    if (verifyAttempts === 1) {
      await redis.expire(verifyRateLimitKey, 600); // 10 minutes
    }
    
    if (verifyAttempts > 5) {
      return res.status(429).json({
        error: {
          code: 'TOO_MANY_ATTEMPTS',
          message: 'Too many failed attempts. Please request a new OTP.',
          retryAfter: 600
        }
      });
    }
    
    // Verify with Auth API
    const authResult = await authApiClient.verifyOTP({ phone, otp });
    
    if (!authResult.success) {
      logger.warn('OTP verification failed', {
        phone: maskPhone(phone),
        ip: req.ip,
        attempt: verifyAttempts
      });
      
      return res.status(400).json({
        error: {
          code: 'INVALID_OTP',
          message: 'Invalid or expired OTP',
          attemptsRemaining: 5 - verifyAttempts
        }
      });
    }
    
    // Clear rate limit on success
    await redis.del(verifyRateLimitKey);
    
    // Find or create user in Consumer DB
    let user = await User.findOne({ phone });
    
    if (!user) {
      user = await User.create({
        phone,
        authProviderId: authResult.user.id,
        roles: ['consumer'],
        activeProfile: 'personal',
        points: { balance: 0, tier: 'bronze' },
        firstLogin: true,
        hasCompletedOnboarding: false
      });
      
      logger.info('New user created', { userId: user._id, phone: maskPhone(phone) });
    } else {
      user.lastLogin = new Date();
      user.firstLogin = false;
      await user.save();
    }
    
    // Generate JWT tokens
    const accessToken = generateAccessToken(user);
    const refreshToken = generateRefreshToken(user);
    
    // Store refresh token in Redis (token rotation)
    await redis.setex(
      `refresh:${user._id}`,
      30 * 24 * 60 * 60, // 30 days
      refreshToken
    );
    
    // Log successful login
    logger.info('User logged in', {
      userId: user._id,
      phone: maskPhone(phone),
      ip: req.ip,
      userAgent: req.get('user-agent')
    });
    
    return res.json({
      accessToken,
      refreshToken,
      user: user.toJSON()
    });
    
  } catch (error) {
    logger.error('OTP verification error', { error: error.message });
    return res.status(500).json({
      error: {
        code: 'VERIFICATION_FAILED',
        message: 'Failed to verify OTP. Please try again.'
      }
    });
  }
};
```

## JWT Implementation

### Token Structure

#### Access Token (24h expiry)

```javascript
const jwt = require('jsonwebtoken');

function generateAccessToken(user) {
  const payload = {
    userId: user._id.toString(),
    phone: user.phone,
    roles: user.roles,
    activeProfile: user.activeProfile,
    type: 'access'
  };
  
  return jwt.sign(
    payload,
    process.env.JWT_SECRET,
    { 
      expiresIn: '24h',
      issuer: 'appzap-consumer-api',
      audience: 'appzap-mobile-app'
    }
  );
}
```

#### Refresh Token (30d expiry)

```javascript
function generateRefreshToken(user) {
  const payload = {
    userId: user._id.toString(),
    type: 'refresh',
    tokenId: uuidv4() // Unique token ID for rotation
  };
  
  return jwt.sign(
    payload,
    process.env.JWT_REFRESH_SECRET,
    { 
      expiresIn: '30d',
      issuer: 'appzap-consumer-api',
      audience: 'appzap-mobile-app'
    }
  );
}
```

### JWT Middleware

```javascript
// middleware/auth.middleware.js
const jwt = require('jsonwebtoken');
const User = require('../models/User');

exports.authenticate = async (req, res, next) => {
  try {
    // Extract token from header
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      return res.status(401).json({
        error: {
          code: 'MISSING_TOKEN',
          message: 'Authentication token is required'
        }
      });
    }
    
    const token = authHeader.substring(7);
    
    // Verify token
    let decoded;
    try {
      decoded = jwt.verify(token, process.env.JWT_SECRET, {
        issuer: 'appzap-consumer-api',
        audience: 'appzap-mobile-app'
      });
    } catch (jwtError) {
      if (jwtError.name === 'TokenExpiredError') {
        return res.status(401).json({
          error: {
            code: 'TOKEN_EXPIRED',
            message: 'Authentication token has expired',
            expiredAt: jwtError.expiredAt
          }
        });
      }
      
      return res.status(401).json({
        error: {
          code: 'INVALID_TOKEN',
          message: 'Invalid authentication token'
        }
      });
    }
    
    // Check token type
    if (decoded.type !== 'access') {
      return res.status(401).json({
        error: {
          code: 'INVALID_TOKEN_TYPE',
          message: 'Invalid token type'
        }
      });
    }
    
    // Get user from database
    const user = await User.findById(decoded.userId);
    
    if (!user || user.isDeleted) {
      return res.status(401).json({
        error: {
          code: 'USER_NOT_FOUND',
          message: 'User not found'
        }
      });
    }
    
    // Attach user to request
    req.user = {
      id: user._id.toString(),
      phone: user.phone,
      roles: user.roles,
      activeProfile: user.activeProfile,
      fullUser: user
    };
    
    // Add request ID for tracing
    req.requestId = uuidv4();
    
    next();
    
  } catch (error) {
    logger.error('Authentication error', { error: error.message });
    return res.status(500).json({
      error: {
        code: 'AUTH_ERROR',
        message: 'Authentication failed'
      }
    });
  }
};
```

### Optional Authentication (for public endpoints)

```javascript
exports.optionalAuth = async (req, res, next) => {
  try {
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith('Bearer ')) {
      req.user = null;
      return next();
    }
    
    // Try to authenticate, but don't fail if invalid
    const token = authHeader.substring(7);
    const decoded = jwt.verify(token, process.env.JWT_SECRET, { ignoreExpiration: false });
    
    if (decoded.type === 'access') {
      const user = await User.findById(decoded.userId);
      if (user && !user.isDeleted) {
        req.user = {
          id: user._id.toString(),
          phone: user.phone,
          roles: user.roles,
          activeProfile: user.activeProfile
        };
      }
    }
  } catch (error) {
    // Silently fail, user stays null
  }
  
  next();
};
```

### Refresh Token Endpoint

```javascript
exports.refreshToken = async (req, res) => {
  try {
    const { refreshToken } = req.body;
    
    if (!refreshToken) {
      return res.status(400).json({
        error: {
          code: 'MISSING_REFRESH_TOKEN',
          message: 'Refresh token is required'
        }
      });
    }
    
    // Verify refresh token
    let decoded;
    try {
      decoded = jwt.verify(refreshToken, process.env.JWT_REFRESH_SECRET);
    } catch (jwtError) {
      return res.status(401).json({
        error: {
          code: 'INVALID_REFRESH_TOKEN',
          message: 'Invalid or expired refresh token'
        }
      });
    }
    
    // Check if refresh token is in Redis (not revoked)
    const storedToken = await redis.get(`refresh:${decoded.userId}`);
    
    if (storedToken !== refreshToken) {
      // Token has been rotated or revoked
      return res.status(401).json({
        error: {
          code: 'TOKEN_REVOKED',
          message: 'Refresh token has been revoked'
        }
      });
    }
    
    // Get user
    const user = await User.findById(decoded.userId);
    
    if (!user || user.isDeleted) {
      return res.status(401).json({
        error: {
          code: 'USER_NOT_FOUND',
          message: 'User not found'
        }
      });
    }
    
    // Generate new tokens (token rotation)
    const newAccessToken = generateAccessToken(user);
    const newRefreshToken = generateRefreshToken(user);
    
    // Store new refresh token, delete old one
    await redis.setex(
      `refresh:${user._id}`,
      30 * 24 * 60 * 60,
      newRefreshToken
    );
    
    logger.info('Token refreshed', { userId: user._id });
    
    return res.json({
      accessToken: newAccessToken,
      refreshToken: newRefreshToken
    });
    
  } catch (error) {
    logger.error('Token refresh error', { error: error.message });
    return res.status(500).json({
      error: {
        code: 'REFRESH_FAILED',
        message: 'Failed to refresh token'
      }
    });
  }
};
```

## Authorization & Permissions

### Role-Based Access Control (RBAC)

```javascript
// middleware/auth.middleware.js

exports.requireRole = (...allowedRoles) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        error: {
          code: 'UNAUTHORIZED',
          message: 'Authentication required'
        }
      });
    }
    
    const hasRole = allowedRoles.some(role => req.user.roles.includes(role));
    
    if (!hasRole) {
      return res.status(403).json({
        error: {
          code: 'FORBIDDEN',
          message: 'Insufficient permissions',
          requiredRoles: allowedRoles,
          userRoles: req.user.roles
        }
      });
    }
    
    next();
  };
};

// Usage in routes:
router.post('/users/merchant/link', 
  authenticate, 
  requireRole('consumer'),
  merchantController.linkMerchant
);
```

### Profile-Based Authorization

```javascript
exports.requireProfile = (requiredProfile) => {
  return (req, res, next) => {
    if (!req.user) {
      return res.status(401).json({
        error: { code: 'UNAUTHORIZED', message: 'Authentication required' }
      });

    if (req.user.activeProfile !== requiredProfile) {
      return res.status(403).json({
        error: {
          code: 'INVALID_PROFILE',

        }
      });
    }
    
    next();
  };
};

);
```

### Resource Ownership Validation

```javascript
exports.validateOwnership = (resourceType) => {
  return async (req, res, next) => {
    try {
      const resourceId = req.params.id || req.params.orderId || req.params.cartId;
      
      let resource;
      switch (resourceType) {
        case 'order':
          resource = await Order.findById(resourceId);
          break;
        case 'cart':
          resource = await Cart.findById(resourceId);
          break;
        case 'booking':
          resource = await Booking.findById(resourceId);
          break;
        default:
          return res.status(500).json({ error: { code: 'INVALID_RESOURCE_TYPE' } });
      }
      
      if (!resource) {
        return res.status(404).json({
          error: { code: 'RESOURCE_NOT_FOUND', message: `${resourceType} not found` }
        });
      }
      
      if (resource.userId.toString() !== req.user.id) {
        return res.status(403).json({
          error: {
            code: 'ACCESS_DENIED',
            message: `You don't have permission to access this ${resourceType}`
          }
        });
      }
      
      req.resource = resource;
      next();
      
    } catch (error) {
      logger.error('Ownership validation error', { error: error.message });
      return res.status(500).json({ error: { code: 'VALIDATION_FAILED' } });
    }
  };
};

// Usage:
router.delete('/eats/cart/:id', 
  authenticate, 
  validateOwnership('cart'),
  cartController.delete
);
```

## Webhook Security

### Phapay Webhook Verification

```javascript
// middleware/webhook.middleware.js
const crypto = require('crypto');

exports.verifyPhapaySignature = (req, res, next) => {
  try {
    const signature = req.headers['x-phapay-signature'];
    
    if (!signature) {
      logger.warn('Webhook received without signature', { ip: req.ip });
      return res.status(401).json({
        error: { code: 'MISSING_SIGNATURE', message: 'Webhook signature required' }
      });
    }
    
    // Compute expected signature
    const payload = req.rawBody; // Raw body string
    const secret = process.env.PHAPAY_WEBHOOK_SECRET;
    
    const expectedSignature = crypto
      .createHmac('sha256', secret)
      .update(payload)
      .digest('hex');
    
    // Constant-time comparison to prevent timing attacks
    if (!crypto.timingSafeEqual(
      Buffer.from(signature),
      Buffer.from(expectedSignature)
    )) {
      logger.warn('Invalid webhook signature', { 
        ip: req.ip,
        receivedSignature: signature.substring(0, 10) + '...'
      });
      
      return res.status(401).json({
        error: { code: 'INVALID_SIGNATURE', message: 'Invalid webhook signature' }
      });
    }
    
    // Verify timestamp to prevent replay attacks (within 5 minutes)
    const timestamp = req.headers['x-phapay-timestamp'];
    const currentTime = Math.floor(Date.now() / 1000);
    
    if (Math.abs(currentTime - timestamp) > 300) {
      logger.warn('Webhook timestamp too old', { timestamp, currentTime });
      return res.status(401).json({
        error: { code: 'EXPIRED_TIMESTAMP', message: 'Webhook timestamp expired' }
      });
    }
    
    // Check for duplicate webhook (idempotency)
    const webhookId = req.body.id || req.body.paymentId;
    const idempotencyKey = `webhook:${webhookId}`;
    
    const exists = await redis.get(idempotencyKey);
    if (exists) {
      // Webhook already processed, return success to avoid retries
      logger.info('Duplicate webhook detected', { webhookId });
      return res.json({ received: true });
    }
    
    // Mark webhook as received (expires in 24 hours)
    await redis.setex(idempotencyKey, 86400, '1');
    
    next();
    
  } catch (error) {
    logger.error('Webhook verification error', { error: error.message });
    return res.status(500).json({
      error: { code: 'VERIFICATION_FAILED', message: 'Failed to verify webhook' }
    });
  }
};

// Usage:
router.post('/v1/payments/webhook/phapay',
  webhookLimiter,
  verifyPhapaySignature,
  paymentController.handleWebhook
);
```

```javascript
// utils/encryption.js
const crypto = require('crypto');

const ALGORITHM = 'aes-256-gcm';
const KEY = Buffer.from(process.env.ENCRYPTION_KEY, 'hex'); // 32 bytes

exports.encrypt = (text) => {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv(ALGORITHM, KEY, iv);
  
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  
  const authTag = cipher.getAuthTag();
  
  return {
    encrypted,
    iv: iv.toString('hex'),
    authTag: authTag.toString('hex')
  };
};

exports.decrypt = (encrypted, iv, authTag) => {
  const decipher = crypto.createDecipheriv(
    ALGORITHM,
    KEY,
    Buffer.from(iv, 'hex')
  );
  
  decipher.setAuthTag(Buffer.from(authTag, 'hex'));
  
  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  
  return decrypted;
};

// Usage in User model:
UserSchema.pre('save', async function(next) {

    const { encrypted, iv, authTag } = encrypt(this.email);
    this.encryptedEmail = encrypted;
    this.emailIV = iv;
    this.emailAuthTag = authTag;
    this.email = undefined; // Don't store plaintext
  }
  next();
});
```

```javascript
// middleware/security.middleware.js

// Detect SQL injection attempts
exports.detectSQLInjection = (req, res, next) => {
  const sqlPattern = /(\b(SELECT|INSERT|UPDATE|DELETE|DROP|CREATE|ALTER|EXEC|UNION)\b)/gi;
  
  const checkValue = (value) => {
    if (typeof value === 'string' && sqlPattern.test(value)) {
      logSecurityEvent('SQL_INJECTION_ATTEMPT', {
        ip: req.ip,
        url: req.originalUrl,
        value: value.substring(0, 100)
      });
      return true;
    }
    return false;
  };
  
  // Check query params
  for (const key in req.query) {
    if (checkValue(req.query[key])) {
      return res.status(400).json({ error: { code: 'INVALID_INPUT' } });
    }
  }
  
  // Check body
  for (const key in req.body) {
    if (checkValue(req.body[key])) {
      return res.status(400).json({ error: { code: 'INVALID_INPUT' } });
    }
  }
  
  next();
};

// Detect XSS attempts
exports.detectXSS = (req, res, next) => {
  const xssPattern = /<script|javascript:|onerror=|onclick=/gi;
  
  const checkValue = (value) => {

      logSecurityEvent('XSS_ATTEMPT', {
        ip: req.ip,
        url: req.originalUrl,
        value: value.substring(0, 100)
      });
      return true;
    }
    return false;
  };
  
  for (const key in req.body) {
    if (checkValue(req.body[key])) {
      return res.status(400).json({ error: { code: 'INVALID_INPUT' } });
    }
  }
```
