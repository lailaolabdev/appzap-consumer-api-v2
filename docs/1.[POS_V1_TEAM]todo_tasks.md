
### Description:
Add a new field `consumerId` to the bill schema. This allows the Consumer API to track which user placed each order.

### Schema Change:
```javascript
// Add to bill schema in bill.model.js:

consumerId: {
  type: String,
  index: true,
  default: null,
  description: 'Consumer API user ID for order tracking'

```

### Why This is Needed:
- Consumer API sends the Consumer user ID when creating bills
- This links POS V1 orders to Consumer API users
- Enables order history feature in the mobile app
- Required for loyalty points and CRM features

## 📝 **TASK 3: Add Endpoint to Get Bills by consumerId**

### Priority: 🟡 **HIGH**

### File to Modify:
```
appzap-app-api/api/src/routes/bill.routes.js
appzap-app-api/api/src/controllers/bill.controller.js
```

### New Endpoint:
```
GET /v3/bills/consumer/:consumerId
```

### Query Parameters:
| Parameter | Type   | Default | Description                |
|-----------|--------|---------|----------------------------|
| skip      | number | 0       | Pagination offset          |
| limit     | number | 20      | Number of bills to return  |
| status    | string | -       | Filter by bill status      |
| startDate | string | -       | Filter by date (ISO 8601)  |
| endDate   | string | -       | Filter by date (ISO 8601)  |

### Response:
```json
{
  "success": true,
  "data": [
    {
      "_id": "bill_id",
      "billNo": "B001",
      "storeId": "store_id",
      "consumerId": "consumer_id",
      "status": "checkout",
      "totalAmount": 150000,
      "createdAt": "2026-01-11T10:30:00Z"
    }
  ],
  "total": 45,
  "skip": 0,
  "limit": 20
}
```

### Implementation:

**1. Add route (bill.routes.js):**
```javascript
// Add this route:
app.get("/v3/bills/consumer/:consumerId", controller.billsByConsumerId);
```

**2. Add controller (bill.controller.js):**
```javascript
/**
 * Get bills by Consumer API user ID
 * GET /v3/bills/consumer/:consumerId
 */
exports.billsByConsumerId = async (req, res) => {
  try {
    const { consumerId } = req.params;
    const { 
      skip = 0, 
      limit = 20, 
      status,
      startDate,
      endDate 
    } = req.query;

    // Build query
    const query = { consumerId };
    
    if (status) {
      query.status = status;
    }
    
    if (startDate || endDate) {
      query.createdAt = {};
      if (startDate) query.createdAt.$gte = new Date(startDate);
      if (endDate) query.createdAt.$lte = new Date(endDate);
    }

    // Execute query
    const bills = await Bill.find(query)
      .sort({ createdAt: -1 })
      .skip(parseInt(skip))
      .limit(parseInt(limit))
      .populate('storeId', 'name image')
      .lean();

    const total = await Bill.countDocuments(query);

    res.json({
      success: true,
      data: bills,
      total,
      skip: parseInt(skip),
      limit: parseInt(limit)
    });

  } catch (error) {
    res.status(500).json({
      success: false,
      error: error.message
    });
  }
};
```

### File to Create:


### Description:
If POS V1 API is exposed to the internet, add API key authentication for Consumer API requests.

### Implementation:
```javascript
// verifyConsumerApiKey.js
const CONSUMER_API_KEY = process.env.CONSUMER_API_KEY;

module.exports = (req, res, next) => {
  // Skip if no API key is configured (internal network)
  if (!CONSUMER_API_KEY) {
    return next();
  }

  const apiKey = req.headers['x-api-key'];
  
  if (!apiKey || apiKey !== CONSUMER_API_KEY) {
    return res.status(401).json({
      success: false,
      error: 'Invalid or missing API key'
    });
  }
  
  next();
};
```

### Environment Variable:
```bash
# Add to .env
CONSUMER_API_KEY=your-secure-api-key-here
```

### Apply to Routes:
```javascript
// In app.js or routes file

// Apply to specific routes
app.get('/v3/stores', verifyConsumerApiKey, controller.stores);
// ... etc
```

## 📝 **TASK 5: Webhook Notifications (Optional)**

### Priority: 🟢 **LOW** (Nice to have for real-time order tracking)

### Description:
When a bill status changes (e.g., from "confirmed" to "cooking" to "served"), send a webhook notification to the Consumer API.

### Consumer API Webhook Endpoint:
```
POST https://consumer-api.appzap.la/api/v1/webhooks/pos-v1
```

### Webhook Payload:
```json
{
  "event": "bill.status_changed",
  "timestamp": "2026-01-11T10:35:00Z",
  "data": {
    "billId": "bill_id_here",
    "consumerId": "consumer_id_here",
    "storeId": "store_id_here",
    "storeName": "Restaurant Name",
    "status": "cooking",
    "previousStatus": "confirmed",
    "totalAmount": 150000
  },
  "signature": "hmac_sha256_signature"
}
```

### Implementation:
```javascript
// Add to bill status update logic
const axios = require('axios');

const notifyConsumerApi = async (billId, newStatus, previousStatus) => {
  const bill = await Bill.findById(billId)
    .populate('storeId', 'name');
  
  if (!bill.consumerId) return; // Skip if no consumer linked

  try {
    await axios.post(process.env.CONSUMER_API_WEBHOOK_URL, {
      event: 'bill.status_changed',
      timestamp: new Date().toISOString(),
      data: {
        billId: bill._id,
        consumerId: bill.consumerId,
        storeId: bill.storeId._id,
        storeName: bill.storeId.name,
        status: newStatus,
        previousStatus,
        totalAmount: bill.totalAmount
      }
    }, {
      headers: {
        'x-webhook-secret': process.env.CONSUMER_API_WEBHOOK_SECRET
      },
      timeout: 5000
    });
  } catch (error) {
    console.error('Failed to notify Consumer API:', error.message);
    // Don't throw - webhook failure shouldn't block status update
  }
};
```
